\section*{Problem 3: Standard error of a proportion  \textit{(5 points)}} % ps 5 24
An article reports that when each football helmet in a random sample of 37 suspension-type helmets was subjected to a
certain impact test, 24 showed damage. Let $p$ denote the proportion of all helmets of this type that would show damage
when tested in the prescribed manner.

\begin{enumerate}[\bf (a)] 
\item  Estimate the value of the proportion $p$. \pts{1}

    \begin{solution}
    \begin{equation*}
      p = \fr{24}{37} \approx \boxed{0.65}
    \end{equation*}
  \end{solution}
  
  \vspace{5ex}
 
\item Compute the standard error of the estimate $\hat p$.\pts{2}
  \begin{solution}
    \begin{align*}
      SE(\hat p) &= \sqrt{\fr{p(1-p)}{n}} = \sqrt{\fr{0.65(0.35)}{37}} \\
                 &=\boxed{0.078}
    \end{align*}
  \end{solution}
  \vspace{5ex}

\item Using the 68–95–99.7 (empirical) rule, estimate the interval that would contain the sample proportion
  $\hat p$ in 95\% of all possible samples of a similar size. \pts{2}

  \begin{solution}
According to the 68–95–99.7 empirical rule, approximately 95\% of all sample proportions will lie within $\pm 2$ standard errors of the population proportion $p$.

\[
\hat p \pm 2\,SE(\hat p)
\]

Substitute the known values:
\[
\hat p = 0.65, \quad SE(\hat p) = 0.078
\]

\[
0.65 \pm 2(0.078) = 0.65 \pm 0.156
\]

\[
\boxed{(0.494, 0.806)}
\]


\end{solution}

 \end{enumerate}

\section*{Problem 5: Estimating probability from a random sample \textit{(8 points)}} %ps 5 24

            \begin{align*}
                np &\geq 10 \\
                n(1 - p) &\geq 10
            \end{align*}

            \begin{align*}
                np &= 100 \times 0.30 = \boxed{30 > 10}\\
                n(1 - p) &= 100 \times 0.70 = \boxed{70 > 10}
            \end{align*}
            Since \boxed{\textbf{both values are greater than 10}}, the Success-Failure Condition is satisfied.
        \end{solution}
       \vspace{5ex}

\section*{Problem 6: Confidence interval of a proportion \textit{(6 points)}}
Recall the example of the solar energy poll we examined in class, with $p=0.88$ and $n=1000$.

  \begin{enumerate}[\bf (a)]
  \item Compute the standard error $SE_{\hat p}$. \pts{2}
        \begin{solution}
        % To compute the standard error \( SE_{\hat{p}} \), we use the formula for the standard error of a proportion:
        \begin{align*}
        SE_{\hat{p}} &= \sqrt{\frac{\hat{p}(1 - \hat{p})}{n}} \\
        &= \sqrt{\frac{0.88 \times (1 - 0.88)}{1000}} \\
        &= \sqrt{\frac{0.88 \times 0.12}{1000}} \\
        &= \sqrt{\frac{0.1056}{1000}} \\
        &= \sqrt{0.0001056} \\
        &\approx 0.0103.
        \end{align*}
        Thus, the standard error is:
        \[
        \boxed{SE_{\hat{p}} \approx 0.0103}
        \]
        \end{solution}
       \vspace{5ex}

       
\item Find the 95\% confidence interval of $\hat p$. (Express your final answer as an interval.) \pts{2}
   \begin{solution}
    For a 95\% confidence level, the critical z-score \( z^* \) is 1.96. 
    %
    The confidence interval is given by:
    \begin{align*}
    CI &= \hat{p} \pm z^* \times SE_{\hat{p}} \\
    &= 0.88 \pm 1.96 \times 0.0103 \\
    &= 0.88 \pm 0.0202.
    \end{align*}
    This gives the interval:
    \[
    \boxed{(0.8598, 0.9002)}
    \]
    \end{solution}
