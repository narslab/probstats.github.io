  function h1(\x, \lx) { return (9*\lx + 3*((\lx)^2) + ((\lx)^3)/3 + 9); };

  function skewnorm(\x, \l) {
    \x = (\l < 0) ? -\x : \x;
    \l = abs(\l);
    \e = exp(-(\x^2)/2);
    return (\l == 0) ? 1 / sqrt(2 * pi) * \e: (
      (\x < -3/\l) ? 0 : (
      (\x < -1/\l) ? \e / (8 * sqrt(2 * pi)) * h1(\x, \x*\l) : (
      (\x <  1/\l) ? \e / (4 * sqrt(2 * pi)) * h2(\x, \x*\l) : (
      (\x <  3/\l) ? \e / (8 * sqrt(2 * pi)) * h3(\x, \x*\l) : (
      sqrt(2/pi) * \e)))));
  };
}

  \section*{Problem 2 \quad {\it Confidence interval of a proportion (10 points)}      }
  A poll conducted in 2013 found that 52\% of U.S. adult X (formerly Twitter) users get at least some news on X. The standard error for this estimate was 2.4\%, and a normal distribution may be used to model the sample proportion $\hat p$. Now answer the following step-by-step questions in order to   construct a 95\% confidence interval for the fraction of U.S. adult X users who get some news on X.

  
  \begin{enumerate}[\bf (a)]
    \item State the value of $\hat p$. \pts{1}
    \begin{solution}
      \begin{align*}
        \hat{p} &= 0.52
      \end{align*}
    \end{solution}
    
    \item Find the corresponding critical $Z$-score for a 95\% confidence level. Show how you obtained your answer (wehther its via Matlab/Python or another source.)\pts{1}
    \begin{solution}
    \begin{align*}
        z &= \texttt{norm.ppf(1 - (1 - 0.95) / 2)} \\
        z &\approx 1.96
    \end{align*}
    \end{solution}


    \item Compute the margin of error. \pts{2}
    \begin{solution}
      \begin{align*}
        \text{Margin of Error} (ME) &= z \times \text{SE} \\
        &= 1.96 \times 0.024 \\
        &= 0.04704
      \end{align*}
    \end{solution}

    
    \item Write the 95\% confidence interval of $p$. \pts{2}
    \begin{solution}
    The Confidence Interval is given by:
      \begin{align*}
        <p>_{0.95} &= \hat{p} \pm \text{ME} \\
        &= 0.52 \pm 0.047 \\
        &= (0.473, 0.567)
      \end{align*}
    \end{solution}

    
    \item Briefly interpret the interval in the context of this question. \pts{2}
    \begin{solution}
      We are 95\% confident that the true proportion of U.S. adult X users who get at least some news on X lies between 47.3\% and 56.7\%.
    \end{solution}

    \item Would you expect the 99\% confidence interval to be narrower (smaller) or wider (larger)? Provide a reason for your answer. \pts{2}
    \begin{solution}
      The 99\% confidence interval would be wider, as increasing the confidence level requires a larger margin of error.
    \end{solution}
  \end{enumerate}

\section*{Problem 3 \quad {\it CI of proportion and sample size (10 points)}      }
An article reports that when each football helmet in a random sample of 37 suspension-type helmets was subjected to a certain impact test, 24 showed damage. Let $p$ denote the proportion of all helmets of this type that would show damage when tested in the prescribed manner.

\begin{enumerate}[\bf (a)]

\item  Calculate a 99\% CI (confidence interval) for $p$. \pts{6}
\begin{solution}
      \begin{align*}
        \hat{p} &= \frac{24}{37} \approx 0.6486 \\
        z &= 2.576 \\
        SE &= \sqrt{\frac{\hat{p} (1 - \hat{p})}{n}} \\
        &= \sqrt{\frac{0.6486 \times 0.3514}{37}} \approx 0.07848 \\
        \text{ME} &= z \times SE \\
        &= 2.576 \times 0.07848 \approx 0.2022 \\
        \therefore <p>_{0.99} &= \hat{p} \pm \text{ME} \\
        &= 0.6486 \pm 0.2022 \\
        & = (0.4464, 0.8508)
      \end{align*}
    \end{solution}

\item What sample size would be required for the width of a 99\% CI to be  0.1? \pts{4}
    \begin{solution}
      \begin{align*}
        \text{Width} = 2 \times \text{ME}  &= 0.1\\
          \therefore \text{ME} &= 0.05 \\
        z &= 2.576 \\
        \hat{p} & \approx 0.6486 \\
        \\
        \text{ME} &= z \times SE \\
        %
        0.05 &= 2.576 \times \sqrt{\frac{0.6486 \times 0.3514}{n}} \\
        n &\approx 605
      \end{align*}
    \end{solution}
    
\end{enumerate}

 \section*{Problem 5 \quad {\it Two-tailed hypothesis test using critical values (10 points)}      }
 400 students were randomly sampled from a large university, and 289 said they did not get enough sleep. Conduct a hypothesis test to check whether this represents a statistically significant difference from 50\%, and use a significance level of 0.05.
 In this problem, you are required to use the critical value appreoach (so no need to compute p-values). Your response will be graded on the following steps:
 \begin{compactitem}
  \item State the hypotheses (there are two) \pts{2}
  \item Compute test statistic \pts{2}
  \item Compute the critical values \pts{2}
  \item Compare test statistic to critical values \pts{2}
  \item State outcome of test and write concluding statement \pts{2}
 \end{compactitem}

\begin{enumerate}

  \item State the hypotheses.
    \begin{solution}
      \begin{align*}
        H_0 &: p = 0.5 \\
        H_1 &: p \neq 0.5
      \end{align*}
    \end{solution}

  \item Compute the test statistic.
    \begin{solution}
      \begin{align*}
        \hat{p} &= \frac{289}{400} = 0.7225 \\
        SE &= \sqrt{\frac{p_0 (1 - p_0)}{n}} = \sqrt{\frac{0.5 \times 0.5}{400}} = 0.025 \\
        z &= \frac{\hat{p} - p_0}{SE} = \frac{0.7225 - 0.5}{0.025} = 8.9
      \end{align*}
    \end{solution}

  \item Compute the critical values.
    % \begin{solution}
    %   \begin{align*}
    %     \text{Critical values} &= \pm 1.96 \quad (\alpha = 0.05, \text{ two-tailed test})
    %   \end{align*}
    % \end{solution}
        \begin{solution}
      Since this is a two-tailed test with a significance level of $\alpha = 0.05$, we split $\alpha$ into two tails (0.025 in each tail).

      We need the z-scores that correspond to the upper and lower 2.5\% of the normal distribution. These z-scores are found using the inverse cumulative distribution function, which gives:
      \begin{align*}
        z_{\alpha/2} &= \texttt{norm.ppf(0.975)} \\
                     &\approx 1.96
      \end{align*}
      
      Thus, the critical values for this test are:
      \begin{align*}
        z &= \pm 1.96
      \end{align*}
    \end{solution}

  \item Compare test statistic to critical values.
    \begin{solution}
      \begin{align*}
        \text{The calculated } z \text{-score is } 8.9, &\text{ which is greater than the critical value of } 1.96. \\
        \text{Since } 8.9 &> 1.96, \text{ we fall within the rejection region.}
      \end{align*}
    \end{solution}


  \item State outcome of test and write concluding statement.
    \begin{solution}
Outcome: We reject $H_0$. 
\\
Conclusion: At the 5\% significance level, there is sufficient evidence to conclude 

that the proportion of students who report not getting enough sleep is significantly different from 50\%.
    \end{solution}
  \end{enumerate}

 \section*{Problem 6 \quad {\it Two-tailed hypothesis test using p-value (12 points)}      }
 It is believed that nearsightedness affects about 8\% of all children. In a random sample of 194 children, 21 are nearsighted. Conduct a hypothesis test ($\alpha=.05$) for the following question: do these data provide evidence that the 8\% value is inaccurate?
 Your response will be graded on the following steps:
 \begin{compactitem}
   \item State the hypotheses (there are two) \pts{2}
   \item Find the standard error \pts{2}
   \item Find the test statistic \pts{2}
   \item Find the p-value \pts{2}
   \item Compare the appropriate values \pts{1}
   \item Clearly state the outcome from your hypothesis test \pts{1}
   \item Write a final concluding statement in response to the question \pts{2}
   \end{compactitem}

\begin{enumerate}

  \item State the hypotheses.
    \begin{solution}
      \begin{align*}
        H_0 &: p = 0.08 \\
        H_1 &: p \neq 0.08
      \end{align*}
    \end{solution}

  \item Find the standard error.
    \begin{solution}
      \begin{align*}
        SE &= \sqrt{\frac{p_0 (1 - p_0)}{n}} = \sqrt{\frac{0.08 \times 0.92}{194}} \approx 0.0195
      \end{align*}
    \end{solution}

  \item Find the test statistic.
    \begin{solution}
      \begin{align*}
        \hat{p} &= \frac{21}{194} \approx 0.1082 \\
        z &= \frac{\hat{p} - p_0}{SE} = \frac{0.1082 - 0.08}{0.0195} \approx \pm 1.45
      \end{align*}
    \end{solution}

  \item Find the p-value.
    \begin{solution}
      \begin{align*}
      \text{p-value} &= 2 \times (1 -\phi(z)) \\
       &= 2 \times (1 -\phi(1.45)) \\
 % normcdf(1.45));

      \end{align*}
    \end{solution}
    
\item Compare the p-value to $\alpha$.
    \begin{solution}
      \begin{align*}
        \alpha &= 0.05 \\
        \therefore \text{p-value} &> \alpha,\\
        \text{Since } 0.147 &> 0.05.%, \text{ we fail to reject } H_0.
      \end{align*}
    \end{solution}

  \item State the outcome of the test.
\begin{solution}
Outcome: We fail to reject $H_0$. \\

\end{solution}

\section*{Problem 7 \quad {\it Difference of two proportions hypothesis testing using p-values (12 points) }      }
A quadcopter company is considering a new supplier for rotor blades. The new supplier would be more expensive, but they claim their higher-quality blades are more reliable,  with 3\% more blades passing inspection than their competitor.
The company's quality control engineer collects a sample of blades, examining 1000 blades from each supplier, and she finds that 899 blades pass inspection from the current supplier and 958 pass inspection from the prospective supplier.
Using these data, set up and evaluate the hypotheses with a significance level $\alpha = .01$.
Your response will be graded on the following steps:
\begin{compactitem}
  \item State the hypotheses (there are two) \pts{2}
  \item Find the standard error \pts{2}
  \item Find the test statistic \pts{2}
  \item Find the p-value \pts{2}
  \item Compare the appropriate values \pts{1}
  \item Clearly state the outcome from your hypothesis test \pts{1}
  \item Write a final concluding statement in response to the question \pts{2}
  \end{compactitem}

\begin{enumerate}

  \item State the hypotheses.
    \begin{solution}
      \textbf{Option 1 (One-tailed test):}
      % \[
      % H_0: p_2 - p_1 = 0.03 \quad \text{vs.} \quad H_1: p_2 - p_1 > 0.03
      % \]
    \begin{align*}
        H_0 &: p_2 - p_1 = 0.03 \\
        H_1 &: p_2 - p_1 > 0.03
    \end{align*}

      \textbf{Option 2 (two-tailed test):}
      % \[
      % H_0: p_2 - p_1 = 0.03 \quad \text{vs.} \quad H_1: p_2 - p_1 \neq 0.03
      % \]
    \begin{align*}
        H_0 &: p_2 - p_1 = 0.03 \\
        H_1 &: p_2 - p_1 \neq 0.03
    \end{align*}

      {\rd \textbf{Note:} Both tests are valid interpretations. The upper-tail test aligns better with testing for a higher pass rate, but the two-sided test is also acceptable. 
      A solution for each option is provided below.}
    \end{solution}

\end{enumerate}

  \item Find the standard error.

           &= \sqrt{0.9285 \times (1 - 0.9285) \times \left( \frac{1}{1000} + \frac{1}{1000} \right)} \\

    \end{solution}

        \\
